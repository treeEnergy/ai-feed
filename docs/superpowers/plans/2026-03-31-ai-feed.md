# AI Feed Implementation Plan

> **For agentic workers:** REQUIRED SUB-SKILL: Use superpowers:subagent-driven-development (recommended) or superpowers:executing-plans to implement this plan task-by-task. Steps use checkbox (`- [ ]`) syntax for tracking.

**Goal:** Build a local Electron desktop app that aggregates AI leaders' content from X, GitHub, Facebook, and arXiv, translates it to Chinese via DeepSeek API, and displays it in an Anthropic-style warm cream UI.

**Architecture:** Electron main process handles scraping (HTTP+Cheerio for arXiv/GitHub, Puppeteer for X/Facebook), translation (DeepSeek API), and storage (SQLite). React renderer with Zustand shows a sidebar of followed people and a timeline feed of translated cards. Scheduled via node-cron.

**Tech Stack:** Electron 33+, React 18, TypeScript, Vite + electron-builder, Tailwind CSS, Zustand, axios + cheerio, puppeteer-core, better-sqlite3, node-cron, DeepSeek API

```typescript
import Database from 'better-sqlite3';
import { v4 as uuid } from 'uuid';

let db: Database.Database;

export function initDatabase(dbPath: string): void {
  db = new Database(dbPath);
  db.pragma('journal_mode = WAL');
  db.pragma('foreign_keys = ON');

  db.exec(`
    CREATE TABLE IF NOT EXISTS persons (
      id TEXT PRIMARY KEY,
      name TEXT NOT NULL,
      title TEXT,
      avatarColor TEXT,
      platforms TEXT DEFAULT '{}',
      isPreset INTEGER DEFAULT 0,
      createdAt TEXT DEFAULT (datetime('now'))
    );

    CREATE TABLE IF NOT EXISTS items (
      id TEXT PRIMARY KEY,
      personId TEXT NOT NULL REFERENCES persons(id),
      platform TEXT NOT NULL,
      originalText TEXT NOT NULL,
      translatedText TEXT,
      url TEXT,
      publishedAt TEXT NOT NULL,
      scrapedAt TEXT DEFAULT (datetime('now')),
      metadata TEXT DEFAULT '{}',
      topics TEXT,
      isRead INTEGER DEFAULT 0,
      isStarred INTEGER DEFAULT 0,
      translationStatus TEXT DEFAULT 'pending'
    );

    CREATE TABLE IF NOT EXISTS weekly_topics (
      weekStart TEXT NOT NULL,
      personId TEXT,
      topics TEXT NOT NULL,
      PRIMARY KEY (weekStart, COALESCE(personId, '__global__'))
    );

    CREATE INDEX IF NOT EXISTS idx_items_person_date ON items(personId, publishedAt DESC);
    CREATE INDEX IF NOT EXISTS idx_items_platform ON items(platform);

  seedPresetPersons();
}
```

```typescript
// electron/storage/queries.ts
import { getDb } from './database';
import { Person, ScrapedItem, Platform, ViewFilter } from '../../src/types';
import { v4 as uuid } from 'uuid';

export class Queries {
  getAllPersons(): Person[] {
    const rows = getDb().prepare("SELECT * FROM persons ORDER BY isPreset DESC, name ASC").all() as any[];
    return rows.map(r => ({ ...r, platforms: JSON.parse(r.platforms), isPreset: !!r.isPreset }));
  }

  addPerson(data: Omit<Person, 'id' | 'createdAt'>): Person {
    const id = uuid();
    getDb().prepare(
      "INSERT INTO persons (id, name, title, avatarColor, platforms, isPreset) VALUES (?, ?, ?, ?, ?, ?)"
    ).run(id, data.name, data.title || '', data.avatarColor, JSON.stringify(data.platforms), data.isPreset ? 1 : 0);
    return { id, ...data, createdAt: new Date().toISOString() };
  }

  deletePerson(id: string): void {
    getDb().prepare("DELETE FROM items WHERE personId = ?").run(id);
    getDb().prepare("DELETE FROM persons WHERE id = ?").run(id);
  }

  getItems(opts: { personId?: string; platform?: Platform; filter?: ViewFilter; limit?: number; offset?: number }): ScrapedItem[] {
    let sql = "SELECT * FROM items WHERE 1=1";
    const params: any[] = [];

    if (opts.personId) { sql += " AND personId = ?"; params.push(opts.personId); }
    if (opts.platform) { sql += " AND platform = ?"; params.push(opts.platform); }
    if (opts.filter === 'starred') { sql += " AND isStarred = 1"; }
    if (opts.filter === 'papers') { sql += " AND platform = 'arxiv'"; }

    sql += " ORDER BY publishedAt DESC";
    sql += ` LIMIT ${opts.limit || 100} OFFSET ${opts.offset || 0}`;

    const rows = getDb().prepare(sql).all(...params) as any[];
    return rows.map(r => ({
      ...r,
      metadata: JSON.parse(r.metadata || '{}'),
      topics: r.topics ? JSON.parse(r.topics) : undefined,
      isRead: !!r.isRead,
      isStarred: !!r.isStarred,
    }));
  }

  upsertItems(items: ScrapedItem[]): void {
    const stmt = getDb().prepare(
      "INSERT OR IGNORE INTO items (id, personId, platform, originalText, translatedText, url, publishedAt, scrapedAt, metadata, topics, isRead, isStarred, translationStatus) VALUES (?, ?, ?, ?, ?, ?, ?, ?, ?, ?, ?, ?, ?)"
    );
    const tx = getDb().transaction(() => {
      for (const item of items) {
        stmt.run(item.id, item.personId, item.platform, item.originalText, item.translatedText || null, item.url, item.publishedAt, item.scrapedAt, JSON.stringify(item.metadata), item.topics ? JSON.stringify(item.topics) : null, item.isRead ? 1 : 0, item.isStarred ? 1 : 0, item.translationStatus);
      }
    });
    tx();
  }

  toggleStar(id: string): void {
    getDb().prepare("UPDATE items SET isStarred = CASE WHEN isStarred = 1 THEN 0 ELSE 1 END WHERE id = ?").run(id);
  }

  markRead(id: string): void {
    getDb().prepare("UPDATE items SET isRead = 1 WHERE id = ?").run(id);
  }

  updateTranslation(id: string, translatedText: string, topics: string[]): void {
    getDb().prepare(
      "UPDATE items SET translatedText = ?, topics = ?, translationStatus = 'done' WHERE id = ?"
    ).run(translatedText, JSON.stringify(topics), id);
  }

  getItemsNeedingTranslation(): ScrapedItem[] {
    return this.getItems({} as any).filter(i => i.translationStatus === 'pending');
  }

  getWeeklyTopics(weekStart?: string, personId?: string): any[] {
    let sql = "SELECT * FROM weekly_topics WHERE 1=1";
    const params: any[] = [];
    if (weekStart) { sql += " AND weekStart = ?"; params.push(weekStart); }
    if (personId) { sql += " AND personId = ?"; params.push(personId); }
    sql += " ORDER BY weekStart DESC";
    const rows = getDb().prepare(sql).all(...params) as any[];
    return rows.map(r => ({ ...r, topics: JSON.parse(r.topics) }));
  }

  computeAndCacheWeeklyTopics(weekStart: string): void {
    const weekEnd = new Date(new Date(weekStart).getTime() + 7 * 24 * 60 * 60 * 1000).toISOString().split('T')[0];
    const items = getDb().prepare(
      "SELECT topics FROM items WHERE publishedAt >= ? AND publishedAt < ? AND topics IS NOT NULL"
    ).all(weekStart, weekEnd) as any[];

    const counts: Record<string, number> = {};
    for (const row of items) {
      const topics = JSON.parse(row.topics) as string[];
      for (const t of topics) { counts[t] = (counts[t] || 0) + 1; }
    }

    const sorted = Object.entries(counts).sort((a, b) => b[1] - a[1]).slice(0, 5).map(([name, count]) => ({ name, count }));

    getDb().prepare(
      "INSERT OR REPLACE INTO weekly_topics (weekStart, personId, topics) VALUES (?, NULL, ?)"
    ).run(weekStart, JSON.stringify(sorted));
  }
```

```typescript
export abstract class BaseScraper implements Scraper {
  abstract platform: string;
  abstract scrape(person: Person): Promise<ScrapedItem[]>;

  protected async withRetry<T>(fn: () => Promise<T>, maxRetries = 3): Promise<T> {
    let lastError: Error | undefined;
    for (let i = 0; i < maxRetries; i++) {
      try {
        return await fn();
      } catch (e) {
        lastError = e as Error;
        if (i < maxRetries - 1) {
          await new Promise(r => setTimeout(r, Math.pow(2, i) * 1000));
        }
      }
    }
    throw lastError;
  }

  protected delay(): Promise<void> {
    return randomDelay();
  }
}
```

```typescript
// electron/translator/deepseek.ts
import axios from 'axios';
import { ScrapedItem } from '../../src/types';

const SYSTEM_PROMPT = `你是一个 AI/ML 领域的专业翻译。请将以下英文内容翻译为流畅的中文。
规则：
1. 保留专业术语原文，如 Transformer、RLHF、AGI、LLM 等
2. 人名保持英文
3. 翻译要自然，不要翻译腔
4. 同时提取 1-2 个关键主题标签（中文）

请返回 JSON 格式：{"translation": "...", "topics": ["...", "..."]}`;

export class DeepSeekTranslator {
  private apiKey: string;
  private baseUrl = 'https://api.deepseek.com/v1';

  constructor(apiKey: string) {
    this.apiKey = apiKey;
  }

  async translate(text: string): Promise<{ translation: string; topics: string[] }> {
    const response = await axios.post(
      `${this.baseUrl}/chat/completions`,
      {
        model: 'deepseek-chat',
        messages: [
          { role: 'system', content: SYSTEM_PROMPT },
          { role: 'user', content: text },
        ],
        response_format: { type: 'json_object' },
        temperature: 0.3,
      },
      {
        headers: {
          Authorization: `Bearer ${this.apiKey}`,
          'Content-Type': 'application/json',
        },
        timeout: 30000,
      },
    );

    return JSON.parse(response.data.choices[0].message.content);
  }

  async translateBatch(items: ScrapedItem[]): Promise<ScrapedItem[]> {
    const results: ScrapedItem[] = [];

    for (const item of items) {
      try {
        const { translation, topics } = await this.translate(item.originalText);
        results.push({
          ...item,
          translatedText: translation,
          topics,
          translationStatus: 'done',
        });
      } catch (e) {
        console.error(`Translation failed for item ${item.id}:`, e);
        results.push({ ...item, translationStatus: 'failed' });
      }
    }

    return results;
  }
}
```

```typescript
// electron/scheduler.ts
import cron from 'node-cron';
import { ScraperEngine } from './scraper';
import { DeepSeekTranslator } from './translator/deepseek';
import { Queries } from './storage/queries';
import { BrowserWindow } from 'electron';
import { ScrapeInterval } from '../src/types';

const INTERVAL_CRON: Record<ScrapeInterval, string> = {
  '30m': '*/30 * * * *',
  '1h': '0 * * * *',
  '2h': '0 */2 * * *',
  '6h': '0 */6 * * *',
};

export class Scheduler {
  private task: cron.ScheduledTask | null = null;
  private isRunning = false;
  private scraperEngine: ScraperEngine;
  private translator: DeepSeekTranslator | null = null;
  private queries: Queries;
  private mainWindow: BrowserWindow | null = null;

  constructor(queries: Queries, apiKey?: string) {
    this.scraperEngine = new ScraperEngine();
    this.queries = queries;
    if (apiKey) {
      this.translator = new DeepSeekTranslator(apiKey);
    }
  }

  setMainWindow(win: BrowserWindow) { this.mainWindow = win; }

  updateApiKey(apiKey: string) {
    this.translator = new DeepSeekTranslator(apiKey);
  }

  start(interval: ScrapeInterval = '2h') {
    this.stop();
    this.task = cron.schedule(INTERVAL_CRON[interval], () => this.run());
  }

  stop() {
    if (this.task) { this.task.stop(); this.task = null; }
  }

  async run(): Promise<void> {
    if (this.isRunning) return;
    this.isRunning = true;

    try {
      this.notify('scrape:status', 'running');

      // 1. Scrape
      const persons = this.queries.getAllPersons();
      const items = await this.scraperEngine.scrapeAll(persons);

      // 2. Dedup & store
      this.queries.upsertItems(items);

      // 3. Translate pending items
      if (this.translator) {
        const pending = this.queries.getItemsNeedingTranslation();
        if (pending.length > 0) {
          const translated = await this.translator.translateBatch(pending);
          for (const item of translated) {
            if (item.translationStatus === 'done' && item.translatedText) {
              this.queries.updateTranslation(item.id, item.translatedText, item.topics || []);
            }
          }
        }
      }

      // 4. Refresh weekly topics
      const monday = this.getCurrentMonday();
      this.queries.computeAndCacheWeeklyTopics(monday);

      // 5. Notify renderer
      this.notify('scrape:status', 'idle');
      this.notify('items:updated', null);
    } catch (e) {
      console.error('Scheduler run failed:', e);
      this.notify('scrape:status', 'error');
    } finally {
      this.isRunning = false;
    }
  }

  private notify(channel: string, data: any) {
    this.mainWindow?.webContents.send(channel, data);
  }

  private getCurrentMonday(): string {
    const d = new Date();
    const day = d.getDay();
    const diff = d.getDate() - day + (day === 0 ? -6 : 1);
    return new Date(d.setDate(diff)).toISOString().split('T')[0];
  }

  async shutdown() {
    this.stop();
    await this.scraperEngine.shutdown();
  }
}
```

```typescript
import { join } from 'path';
import { app } from 'electron';

const DEFAULTS: AppSettings = {
  deepseekApiKey: '',
  scrapeInterval: '2h',
};

function getPath(): string {
  return join(app.getPath('userData'), 'settings.json');
}

export function loadSettings(): AppSettings {
  const p = getPath();
  if (!existsSync(p)) return { ...DEFAULTS };
  try {
    return { ...DEFAULTS, ...JSON.parse(readFileSync(p, 'utf-8')) };
  } catch {
    return { ...DEFAULTS };
  }
}

export function saveSettings(settings: Partial<AppSettings>): AppSettings {
  const current = loadSettings();
  const merged = { ...current, ...settings };
  writeFileSync(getPath(), JSON.stringify(merged, null, 2));
  return merged;
}
```

```typescript
import { Queries } from './storage/queries';

import { loadSettings, saveSettings } from './storage/settings';

export function registerIPCHandlers(queries: Queries, scheduler: Scheduler) {

  ipcMain.handle('items:markRead', (_e, { id }) => queries.markRead(id));

  ipcMain.handle('persons:delete', (_e, { id }) => queries.deletePerson(id));

  ipcMain.handle('settings:get', () => loadSettings());

    if (partial.deepseekApiKey) scheduler.updateApiKey(partial.deepseekApiKey);
    if (partial.scrapeInterval) { scheduler.stop(); scheduler.start(partial.scrapeInterval); }
    return updated;

  ipcMain.handle('weekly-topics:get', (_e, opts) =>

  );

  ipcMain.handle('shell:openExternal', (_e, url: string) => shell.openExternal(url));
```

```typescript
import { useCallback, useEffect } from 'react';

declare global {
  interface Window {
    electronAPI: {
      invoke: (channel: string, ...args: any[]) => Promise<any>;
      on: (channel: string, cb: (...args: any[]) => void) => () => void;
    };
  }
}

export function useIPCInvoke() {
  return useCallback(
    <T = any>(channel: string, ...args: any[]): Promise<T> =>
      window.electronAPI.invoke(channel, ...args),
    []
  );
}

export function useIPCListener(channel: string, callback: (...args: any[]) => void) {
  useEffect(() => {
    const unsub = window.electronAPI.on(channel, callback);
    return unsub;
  }, [channel, callback]);
}
```

```typescript
  fetchItems: () => Promise<void>;

  items: [],

    set({ persons });

    const { selectedPersonId, selectedPlatform, viewFilter } = get();
    set({ isLoading: true });
    const items = await window.electronAPI.invoke('items:getAll', {
      personId: selectedPersonId || undefined,
      platform: selectedPlatform || undefined,
      filter: viewFilter,
      limit: 100,
    });
    set({ items, isLoading: false });
  },
  fetchWeeklyTopics: async () => {
    const weeklyTopics = await window.electronAPI.invoke('weekly-topics:get', {});
```
